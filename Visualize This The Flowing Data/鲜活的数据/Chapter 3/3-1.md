hotdogs <- read.csv("Downloads/hot-dog-contest-winners.csv", sep = ",", header = TRUE)

barplot(hotdogs$Dogs.eaten, names.arg = hotdogs$Year, col = "red", border = NA, xlab = "年份", ylab = "吃掉的HDB数量")

fill_colors <- c()

for (i in 1:length(hotdogs$Country)) {
    if (hotdogs$Country[i] == "United States") {
        fill_colors <- c(fill_colors, "#821122")
    } else {
        fill_colors <- c(fill_colors, "#cccccc")
    }
}

barplot(hotdogs$Dogs.eaten, names.arg = hotdogs$Year, col = fill_colors, border = NA, space = 0.3, xlab = "年份", ylab = "吃掉的HDB数量")

barplot(hotdogs$Dogs.eaten, names.arg = hotdogs$Year, col = fill_colors, border = NA, space = 0.3, xlab = "年份", ylab = "吃掉的HDB数量", main = "1980-2010年\"内森杯\"热狗大胃王比赛成绩")
